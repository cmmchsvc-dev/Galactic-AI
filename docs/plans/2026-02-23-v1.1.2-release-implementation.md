**Step 1: Run the package builder script**

Create and run this Python script (inline, one-shot):

```python
import zipfile, tarfile, hashlib, os, shutil
from pathlib import Path

SRC = Path("F:/Galactic AI Public Release")
OUT = SRC / "releases" / "v1.1.2"
VER = "v1.1.2"
PREFIX = f"Galactic-AI-{VER}"
OUT.mkdir(exist_ok=True)

# Files/dirs to include (same as v1.1.1 but with skills/ instead of plugins/)
INCLUDE_DIRS = [
    "chrome-extension",
    "docs",
    "skills",
]
INCLUDE_FILES = [
    "galactic_core_v2.py", "gateway_v2.py", "web_deck.py",
    "galactic_ai_flux_v4.ico", "flusher.py", "imprint_engine.py",
    "memory_module_v2.py", "model_manager.py", "nvidia_gateway.py",
    "ollama_manager.py", "personality.py", "personality.yaml",
    "remote_access.py", "scheduler.py", "splash.py",
    "discord_bridge.py", "gmail_bridge.py", "telegram_bridge.py",
    "whatsapp_bridge.py",
    "config.yaml", "requirements.txt",
    "README.md", "CHANGELOG.md", "FEATURES.md", "CONTRIBUTING.md",
    "CHROMEBOOK.md", "LICENSE",
    "IDENTITY-example.md", "MEMORY-example.md", "SOUL-example.md",
    "USER-example.md", "VAULT-example.md",
    "index.html", "privacy.html", "terms.html",
]
# Windows-only extras
WIN_EXTRAS = ["install.ps1", "launch.ps1", "update.ps1", "create_shortcut.ps1"]
# Unix extras
UNIX_EXTRAS = ["install.sh", "launch.sh", "update.sh"]

EXCLUDE_SUFFIXES = {".pyc", ".pyo"}
EXCLUDE_DIRS = {"__pycache__", ".git", "logs", "workspace", "releases", "plugins"}

def should_include(path: Path) -> bool:
    for part in path.parts:
        if part in EXCLUDE_DIRS:
            return False
    return path.suffix not in EXCLUDE_SUFFIXES

def collect_files(extras):
    """Collect (src_path, archive_name) pairs."""
    pairs = []
    for f in INCLUDE_FILES + extras:
        p = SRC / f
        if p.exists():
            pairs.append((p, f"{PREFIX}/{f}"))
    for d in INCLUDE_DIRS:
        dp = SRC / d
        if dp.exists():
            for fp in dp.rglob("*"):
                if fp.is_file() and should_include(fp):
                    rel = fp.relative_to(SRC)
                    pairs.append((fp, f"{PREFIX}/{rel.as_posix()}"))
    return pairs

# Windows ZIP
win_path = OUT / f"Galactic-AI-{VER}-windows.zip"
with zipfile.ZipFile(win_path, "w", zipfile.ZIP_DEFLATED) as zf:
    for src, arc in collect_files(WIN_EXTRAS):
        zf.write(src, arc)
print(f"Built: {win_path.name} ({win_path.stat().st_size // 1024} KB)")

# macOS ZIP
mac_path = OUT / f"Galactic-AI-{VER}-macos.zip"
with zipfile.ZipFile(mac_path, "w", zipfile.ZIP_DEFLATED) as zf:
    for src, arc in collect_files(UNIX_EXTRAS):
        zf.write(src, arc)
print(f"Built: {mac_path.name} ({mac_path.stat().st_size // 1024} KB)")

# Generic ZIP
gen_path = OUT / f"Galactic-AI-{VER}.zip"
with zipfile.ZipFile(gen_path, "w", zipfile.ZIP_DEFLATED) as zf:
    for src, arc in collect_files(WIN_EXTRAS + UNIX_EXTRAS):
        zf.write(src, arc)
print(f"Built: {gen_path.name} ({gen_path.stat().st_size // 1024} KB)")

# Linux tar.gz
lin_path = OUT / f"Galactic-AI-{VER}-linux.tar.gz"
with tarfile.open(lin_path, "w:gz") as tf:
    for src, arc in collect_files(UNIX_EXTRAS):
        tf.add(src, arcname=arc)
print(f"Built: {lin_path.name} ({lin_path.stat().st_size // 1024} KB)")

# SHA256SUMS
sha_path = OUT / "SHA256SUMS.txt"
packages = [win_path, mac_path, gen_path, lin_path]
with open(sha_path, "w") as sf:
    for p in packages:
        h = hashlib.sha256(p.read_bytes()).hexdigest()
        sf.write(f"{h}  {p.name}\n")
print(f"Generated: SHA256SUMS.txt")
```
